# Leading spaces must not appear at the beginning of a printed line
>>> text_indentation("   Hello.")
Hello.
<BLANKLINE>

# Spaces after separators must be removed at start of next line
>>> text_indentation("Hi.   There")
Hi.
<BLANKLINE>
There

# Trailing spaces must not appear at the end of the last printed line
>>> text_indentation("End.   ")
End.
<BLANKLINE>
